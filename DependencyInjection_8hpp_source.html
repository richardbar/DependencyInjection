93 class ServiceProvider : public IServiceProvider
94 {
95 private:
96 std::map<std::type_index, std::vector<ServiceDescriptor>> _services;
97
98 explicit ServiceProvider(const std::vector<ServiceDescriptor>& descriptors)
99 {
100 using std::ranges::reverse_view;
101
102 for (const auto& descriptor : reverse_view(descriptors))
103 {
104 const auto& typeInfo = descriptor.GetTypeInfo();
105 bool serviceExists = _services.contains(typeInfo);
106 if (serviceExists)
107 {
108 _services[typeInfo].push_back(descriptor);
109 }
110 else
111 {
112 _services.insert({ typeInfo, { descriptor }});
113 }
114 }
115 }
116
117 public:
118 ServiceProvider() = delete;
119
120 [[nodiscard]] std::any GetService(const std::type_info& type) final
121 {
122 const auto position = _services.find(type);
123 const auto notFound = (position == _services.end());
124 if (notFound)
125 {
126 return {};
127 }
128
129 const auto descriptorList = position->second;
130 const auto& descriptor = descriptorList[0];
131 const auto factory = descriptor.GetFactory();
132 auto service = factory(*this);
133
134 return service;
135 }
