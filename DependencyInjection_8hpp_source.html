146DependencyInjection::ServiceProvider::ServiceProvider(const std::vector<DependencyInjection::ServiceDescriptor>& serviceDescriptors)
147{
148 for (const auto& serviceDescriptor : std::ranges::reverse_view(serviceDescriptors))
149 {
150 bool serviceDoesntExists = (!this->_services.contains(serviceDescriptor.GetTypeInfo()));
151 if (serviceDoesntExists)
152 {
153 this->_services.insert({ serviceDescriptor.GetTypeInfo(), { serviceDescriptor } });
154 }
155 else
156 {
157 this->_services[serviceDescriptor.GetTypeInfo()].push_back(serviceDescriptor);
158 }
159 }
160}

162std::shared_ptr<void> DependencyInjection::ServiceProvider::GetService(const std::type_info& type)
163{
164 auto positionInMap = this->_services.find(type);
165
166 auto notFound = (positionInMap == this->_services.end());
167 if (notFound)
168 {
169 return nullptr;
170 }
171
172 auto serviceDescriptorList = positionInMap->second;
173 auto firstServiceDescriptor = serviceDescriptorList[0];
174 auto serviceFactory = firstServiceDescriptor.GetFactory();
175 auto service = serviceFactory(*this);
176
177 return service;
178}
